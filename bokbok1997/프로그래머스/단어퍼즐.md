# 단어퍼즐

```python
def solution(strs, t):
    n = max(len(s) for s in strs)
    length = len(t)
    dp = [length+1]*(length+1)
    dp[0]=0
    for i in range(1,length+1):
        for l in range(1,n+1):
            if t[i-l:i] in strs:
                dp[i] = min(dp[i-l]+1,dp[i])
    print(dp)
    if dp[-1] == length+1:
        return -1
    else:
        return dp[-1]
```
